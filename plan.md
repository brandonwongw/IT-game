# Bran's Game Center Homepage Implementation Plan

## Project Overview
Create a homepage (`index.html`) for Bran's Game Center with Luffy Gear 5 style theme, featuring navigation to two games: Snake v2.0 and Pokemon.

## Design Specifications
- **Theme**: Luffy's Gear 5 style (playful, energetic, cartoonish)
- **Color Scheme**: 
  - Primary: `#7A55E6` (purple)
  - Secondary: `#F5F5F5` (light gray/white)
  - Accent: `#FFD166` (yellow), `#EF476F` (pink), `#06D6A0` (teal)
- **Header**: "Tuff & Fun"
- **Game Cards**: Each card contains [Game Name / Description / Start Button]
- **Footer**: Copyright info + Brandon Wong

// Render game cards
function renderGameCards() {
    gamesContainer.innerHTML = '';
    
    games.forEach(game => {
        const gameCard = document.createElement('div');
        gameCard.className = 'game-card';
        gameCard.dataset.id = game.id;
        
        gameCard.innerHTML = `
            <div class="game-card-header">
                <div class="game-icon">${game.icon}</div>
                <h3 class="game-title">${game.title}</h3>
                <p class="game-subtitle">${game.subtitle}</p>

                <p class="game-description">${game.description}</p>
                <ul class="game-features">
                    ${game.features.map(feature => `<li>${feature}</li>`).join('')}

// Setup event listeners
function setupEventListeners() {
    // Play button click events
    document.addEventListener('click', function(e) {
        if (e.target.closest('.play-button')) {
            const button = e.target.closest('.play-button');
            const file = button.getAttribute('data-file');
            
            // Add click animation
            button.style.transform = 'scale(0.95)';
            setTimeout(() => {
                button.style.transform = '';
            }, 200);
            
            // Log game start
            console.log(`Starting game: ${file}`);
        }
    });
    
    // Game card hover effects
    const gameCards = document.querySelectorAll('.game-card');
    gameCards.forEach(card => {
        card.addEventListener('mouseenter', function() {
            this.style.zIndex = '10';
        });
        
        card.addEventListener('mouseleave', function() {
            this.style.zIndex = '';
        });
    });
}

// Add some fun animations
function addAnimations() {
    // Add floating animation to game icons
    const gameIcons = document.querySelectorAll('.game-icon');
    gameIcons.forEach((icon, index) => {
        icon.style.animation = `float ${3 + index * 0.5}s ease-in-out infinite alternate`;
    });
    
    // Add staggered animation to feature cards
    const featureCards = document.querySelectorAll('.feature-card');
    featureCards.forEach((card, index) => {
        card.style.animationDelay = `${index * 0.2}s`;
        card.classList.add('animate__animated', 'animate__fadeInUp');
    });
}
